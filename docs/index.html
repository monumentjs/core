Modular Architecture

Each application runs with @Boot decorator.

RealWorldApplication.ts

@Boot
@Application({
    modules: [
        LoggerModule,
        ConfigurationModule
    ]
})
export class RealWorldApplication {
    private readonly _logger: Logger;

    public constructor(loggerManager: LoggerManager) {
        this._logger = loggerManager.getLogger(this.constructor.name);
    }

    @Init
    public async init() {
        await this._logger.info('Hello Real World');
    }
}
modules/configuration/ConfigurationModule.ts

@Module({
    components: [
        LoggerModule,
        LoggingConfiguration
    ]
})
export class ConfigurationModule {
}
modules/configuration/diagnostics/LoggingConfiguration.ts

@Configuration
export class LoggingConfiguration {

    @Unit(LoggerConfiguration)
    public loggerConfiguration(): LoggerConfiguration {
        return new LoggerConfiguration([
            new ConsoleAppender(ConsoleMessageLayout.SIMPLE)
        ]);
    }
}
